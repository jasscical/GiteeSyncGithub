思路：自定义双链表，并结合unordered_map哈希表来实现。

封装性很好。

```c++
class LRUCache {
struct DlinkedNode{
    int key, value;
    DlinkedNode* prev;
    DlinkedNode* next;
    DlinkedNode(): key(0), value(0), prev(nullptr), next(nullptr) {}
    DlinkedNode(int _key, int _value): key(_key), value(_value), prev(nullptr), next(nullptr) {}
};
public:
    LRUCache(int _capacity) {
        // 初始化cache的容量
        // 并初始化两个DlinkedNode节点
        // 互相指向对方
        capacity = _capacity;
        size = 0; // 刷题的时候忘记了这个
        head = new DlinkedNode();
        tail = new DlinkedNode();
        head->next = tail;
        tail->prev = head;
    }
    
    int get(int key) {
        if(cache.count(key) == 0) return -1;
        // key存在，通过哈希定位，再返回值，
        // 然后把节点放到头部
        DlinkedNode* node = cache[key];
        moveToHead(node);
        return node->value;
    }
    
    void put(int key, int value) {
        // 如果key不存在，则需要创建一个节点
        if(cache.count(key) == 0){
            DlinkedNode* node = new DlinkedNode(key, value);
            // 添加进哈希cache
            cache[key] = node;
            // 添加到头部
            addToHead(node);
            ++size;
            if(size > capacity){
                // 超出容量，删除双向链表尾部节点
                DlinkedNode* removed = removeTail();
                // 删除哈希中对应的项
                cache.erase(removed->key);
                // 防止内存泄露
                delete removed;
                --size;
            }
        }
        else{ // 如果key存在
            // 通过哈希定位，然后更新值
            // 再移至头部
            DlinkedNode* node = cache[key];
            node->value = value;
            moveToHead(node);
        }
    }
private:
    unordered_map<int, DlinkedNode*> cache;
    DlinkedNode* head;
    DlinkedNode* tail;
    int capacity, size;
    DlinkedNode* removeTail(){
        DlinkedNode* node = tail->prev;
        removeNode(node);
        return node;
    }
    // moveToHead所需要的操作API
    void addToHead(DlinkedNode* node){
        // head 、node 、head->next
        // 先安排node的指向,再安排head
        node->prev = head;
        node->next = head->next;
        head->next->prev = node;
        head->next = node;
    }
    void removeNode(DlinkedNode* node){
        node->prev->next = node->next;
        node->next->prev = node->prev;
    }
    void moveToHead(DlinkedNode* node){
        removeNode(node); // 先删掉节点
        addToHead(node); // 再移到头部
    }
};

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache* obj = new LRUCache(capacity);
 * int param_1 = obj->get(key);
 * obj->put(key,value);
 */
```
